volatile float rvec [125];
                volatile float rmat [125][125];
                float res [125];
                
                void matVecMul(){
                  for(int i=0; i<125; i++) {
                    res[i] = 0.0;
                    for(int j=0; j<125; j++) {
                      res[i] += rvec[j] * rmat[i][j];    
                    }
                  }
                }
